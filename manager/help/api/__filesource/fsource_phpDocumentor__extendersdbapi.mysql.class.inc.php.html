  229.    private function __insert($insert_method, $fields, $intotable, $fromfields = "*", $fromtable = "", $where = "", $limit = "") {
  230.       if (!$intotable)
  231.          return false;
  232.       else {
  233.          if (!is_array($fields))
  234.             $flds = $fields;
  235.          else {
  236.             $keys = array_keys($fields);
  237.             $values = array_values($fields);
  238.             $flds = "(" . implode(",", $keys) . ") " .
  239.              (!$fromtable && $values ? "VALUES('" . implode("','", $values) . "')" : "");
  240.             if ($fromtable) {
  241.                $where = ($where != "") ? "WHERE $where" : "";
  242.                $limit = ($limit != "") ? "LIMIT $limit" : "";
  243.                $sql = "SELECT $fromfields FROM $fromtable $where $limit";
  244.             }
  245.          }
  246.          $rt = $this->query("$insert_method $intotable $flds $sql");
  247.          $lid = $this->getInsertId();
  248.          return $lid ? $lid : $rt;
  249.       }
  250.    }
